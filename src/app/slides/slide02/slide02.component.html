// 1. Observer
interface BasicObserver {{ '{' }}
  next: (value: any) => void;
}

// 2. Observable (non fonctionnel)
class SimpleObservable {{ '{' }}
  private onsubscribe: (obs: BasicObserver) => void;

  subscribe(obs: BasicObserver) {{ '{' }}
    this.onsubscribe(obs);
  }
}
